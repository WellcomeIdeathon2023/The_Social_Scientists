Remove duplicate tweets and other things


# Removing Duplicate tweets
twitter_data$Duplicate = duplicated(twitter_data$text)
twitter_data = subset(twitter_data, twitter_data$Duplicate=="FALSE")
twitter_data$Duplicate=NULL

# Removing blank spaces, punctuation, links, extra spaces, special characters and other unwanted things.
twitter_data$text = gsub("[:blank:]", "", twitter_data$text) #blank spaces
twitter_data$text = gsub("[[:punct:]]", "", twitter_data$text) #punctuations
#twitter_data$text = gsub("[:cntrl:]", "", twitter_data$text) #special characters
twitter_data$text = gsub("[[:digit:]]", "", twitter_data$text) #numbers
#twitter_data$text = gsub("(RT|via)((?:\\b\\W*@\\w+)+)", " ",  twitter_data$text) #retweet mentions
twitter_data$text <- gsub("@[[:alpha:]]*","", twitter_data$text) #twitter mentions
twitter_data$text = gsub("http\\w+", "", twitter_data$text) #links
